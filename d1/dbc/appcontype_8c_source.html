127 CipConnectionObject *GetIoConnectionForConnectionData(
128  CipConnectionObject *const RESTRICT connection_object,
129  EipUint16 *const extended_error) {
130 
131  *extended_error = 0;
132 
133  CipConnectionObject *io_connection = GetExclusiveOwnerConnection(
134  connection_object,
135  extended_error);
136  if (NULL == io_connection) {
137  if (kConnectionManagerExtendedStatusCodeSuccess == *extended_error) {
138  /* we found no connection and don't have an error so try input only next */
139  io_connection = GetInputOnlyConnection(connection_object, extended_error);
140  if (NULL == io_connection) {
141  if (kConnectionManagerExtendedStatusCodeSuccess == *extended_error) {
142  /* we found no connection and don't have an error so try listen only next */
143  io_connection = GetListenOnlyConnection(connection_object,
144  extended_error);
145  if ( (NULL == io_connection) && (0 == *extended_error) ) {
146  /* no application connection type was found that suits the given data */
147  *extended_error =
148  kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
149  } else {
150  ConnectionObjectSetInstanceType(connection_object,
151  kConnectionObjectInstanceTypeIOListenOnly);
152  OPENER_TRACE_INFO("IO Listen only connection requested\n");
153  //Is listen only connection
154  }
155  }
156  } else {
157  ConnectionObjectSetInstanceType(connection_object,
158  kConnectionObjectInstanceTypeIOInputOnly);
159  OPENER_TRACE_INFO("IO Input only connection requested\n");
160  //is Input only connection
161  }
162  }
163  } else {
164  ConnectionObjectSetInstanceType(connection_object,
165  kConnectionObjectInstanceTypeIOExclusiveOwner);
166  OPENER_TRACE_INFO("IO Exclusive Owner connection requested\n");
167  //Is exclusive owner connection
168  }
169 
170  if (NULL != io_connection) {
171  ConnectionObjectDeepCopy(io_connection, connection_object);
172  }
173 
174  return io_connection;
175 }

177 CipConnectionObject *GetExclusiveOwnerConnection(
178  const CipConnectionObject *const RESTRICT connection_object,
179  EipUint16 *const extended_error) {
180 
181  for (size_t i = 0; i < OPENER_CIP_NUM_EXLUSIVE_OWNER_CONNS; ++i) {
182  if ( (g_exlusive_owner_connections[i].output_assembly ==
183  connection_object->consumed_path.instance_id)
184  && (g_exlusive_owner_connections[i].input_assembly ==
185  connection_object->produced_path.instance_id)
186  && (g_exlusive_owner_connections[i].config_assembly ==
187  connection_object->configuration_path.instance_id) ) {
188 
189  /* check if on other connection point with the same output assembly is currently connected */
190  CipConnectionObject *exclusive_owner = GetConnectedOutputAssembly(
191  connection_object->produced_path.instance_id);
192  if ( NULL
193  != exclusive_owner ) {
194  if(kConnectionObjectStateEstablished ==
195  ConnectionObjectGetState(exclusive_owner) ) {
196  *extended_error =
197  kConnectionManagerExtendedStatusCodeErrorOwnershipConflict;
198  OPENER_TRACE_INFO("Hit an Ownership conflict in appcontype.c");
199  break;
200  }
201  if(kConnectionObjectStateTimedOut ==
202  ConnectionObjectGetState(exclusive_owner)
203  && ConnectionObjectEqualOriginator(connection_object,
204  exclusive_owner) ) {
205  g_exlusive_owner_connections[i].connection_data.
206  connection_close_function(&(g_exlusive_owner_connections[i].
207  connection_data) );
208  return &(g_exlusive_owner_connections[i].connection_data);
209  } else {
210  *extended_error =
211  kConnectionManagerExtendedStatusCodeErrorOwnershipConflict;
212  OPENER_TRACE_INFO(
213  "Hit an Ownership conflict with timed out connection");
214  break;
215  }
216  }
217  return &(g_exlusive_owner_connections[i].connection_data);
218  break;
219  }
220  }
221  return NULL;

224 CipConnectionObject *GetInputOnlyConnection(

227 
228  for (size_t i = 0; i < OPENER_CIP_NUM_INPUT_ONLY_CONNS; ++i) {
229  if (g_input_only_connections[i].output_assembly

231  if (g_input_only_connections[i].input_assembly

233  *extended_error =
234  kConnectionManagerExtendedStatusCodeInvalidProducingApplicationPath;

241  break;
242  }

245  ++j) {

247  == ConnectionObjectGetState(&(g_input_only_connections[i].
248  connection_data[j]) )
249  && ConnectionObjectEqualOriginator(connection_object,
250  &(g_input_only_connections[i].
251  connection_data[j]) ) )
252  {
253  g_input_only_connections[i].connection_data[j].
254  connection_close_function(
255  &g_input_only_connections[i].connection_data[j]);
256  return &(g_input_only_connections[i].connection_data[j]);
257  }

264  connection_data[j]) ) ) {

344 CipConnectionObject *GetExistingProducerMulticastConnection(
345  const EipUint32 input_point) {
346  DoublyLinkedListNode *node = connection_list.first;
347 
348  while (NULL != node) {
349  CipConnectionObject *producer_multicast_connection = node->data;
350  if ( true ==
351  ConnectionObjectIsTypeIOConnection(producer_multicast_connection) &&
352  (input_point ==
353  producer_multicast_connection->produced_path.instance_id) &&
354  ( kConnectionObjectConnectionTypeMulticast ==
355  ConnectionObjectGetTToOConnectionType(producer_multicast_connection) )
356  &&
357  (kEipInvalidSocket !=
358  producer_multicast_connection->socket[
359  kUdpCommuncationDirectionProducing]) )
360  {
361  /* we have a connection that produces the same input assembly,
362  * is a multicast producer and manages the connection.

366  node = node->next;
367  }
