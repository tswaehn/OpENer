00593 
00594   switch (cip_type)
00595   /* check the data type of attribute */
00596   {
00597     case (kCipBool):
00598     case (kCipSint):
00599     case (kCipUsint):
00600     case (kCipByte):
00601       *(EipUint8 *) (data) = **message;
00602       ++(*message);
00603       number_of_decoded_bytes = 1;
00604       break;
00605 
00606     case (kCipInt):
00607     case (kCipUint):
00608     case (kCipWord):
00609       ( *(EipUint16 *) (data) ) = GetIntFromMessage(message);
00610       number_of_decoded_bytes = 2;

00649

00651       *message += short_string->length;

00719 int EncodeEPath(CipEpath *epath, EipUint8 **message) {
00720   unsigned int length = epath->path_size;
00721   AddIntToMessage(epath->path_size, message);
00722 
00723   if (epath->class_id < 256) {
00724     **message = 0x20; /*8Bit Class Id */
00725     ++(*message);
00726     **message = (EipUint8) epath->class_id;
00727     ++(*message);
00728     length -= 1;
00729   } else {
00730     **message = 0x21; /*16Bit Class Id */
00731     ++(*message);
00732     **message = 0; /*pad byte */
00733     ++(*message);
00734     AddIntToMessage(epath->class_id, message);
00735     length -= 2;
00736   }
00737 
00738   if (0 < length) {
00739     if (epath->instance_number < 256) {
00740       **message = 0x24; /*8Bit Instance Id */
00741       ++(*message);
00742       **message = (EipUint8) epath->instance_number;
00743       ++(*message);
00744       length -= 1;
00745     } else {
00746       **message = 0x25; /*16Bit Instance Id */
00747       ++(*message);
00748       **message = 0; /*padd byte */
00749       ++(*message);
00750       AddIntToMessage(epath->instance_number, message);
00751       length -= 2;
00752     }
00753 
00754     if (0 < length) {
00755       if (epath->attribute_number < 256) {
00756         **message = 0x30; /*8Bit Attribute Id */
00757         ++(*message);
00758         **message = (EipUint8) epath->attribute_number;
00759         ++(*message);
00760         length -= 1;
00761       } else {
00762         **message = 0x31; /*16Bit Attribute Id */
00763         ++(*message);

00765         ++(*message);
00766         AddIntToMessage(epath->attribute_number, message);
00767         length -= 2;

00769     }
00770   }
