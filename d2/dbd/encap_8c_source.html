00436   int session_index = 0;
00437   const EipUint8 *receive_data_buffer = NULL;
00438   EipUint16 protocol_version = GetIntFromMessage(

00442

00446     /* check if the socket has already a session open */

00448       if (g_registered_sessions[i] == socket) {
00449         /* the socket has already registered a session this is not allowed*/

00451         receive_data->status = kEncapsulationProtocolInvalidCommand;

00454           &receive_data->communication_buffer_start[
00455             kEncapsulationHeaderSessionHandlePosition];

00457                          (EipUint8 **const)&receive_data_buffer);                               /*EncapsulateData will not update the session handle so we have to do it here by hand*/

00459       }
00460     }
00461
