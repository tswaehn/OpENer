00593   CipCommonPacketFormatData *common_packet_format_data_item,

00595

00597

00599     /* add Interface Handle and Timeout = 0 -> only for SendRRData and SendUnitData necessary */
00600     AddDintToMessage(0, &message);
00601     AddIntToMessage(0, &message);
00602     message_size += 6;

00604

00606                                  message_size);
00607 
00608   /* process Address Item */

00610     case kCipItemIdNullAddress: {
00611       message_size = EncodeNullAddressItem(&message, message_size);
00612       break;
00613     }
00614     case kCipItemIdConnectionAddress: {
00615       message_size = EncodeConnectedAddressItem(&message,
00616                                                 common_packet_format_data_item,
00617                                                 message_size);
00618       break;

00620     case kCipItemIdSequencedAddressItem: {
00621       message_size = EncodeSequencedAddressItem(&message,
00622                                                 common_packet_format_data_item,
00623                                                 message_size);
00624       break;
00625     }
00626   }
00627 
00628   /* process Data Item */
00629   if ( (common_packet_format_data_item->data_item.type_id
00630         == kCipItemIdUnconnectedDataItem)
00631        || (common_packet_format_data_item->data_item.type_id
00632            == kCipItemIdConnectedDataItem) ) {
00633 
00634     if (message_router_response) {
00635       message_size = EncodeDataItemType(common_packet_format_data_item,
00636                                         &message, message_size);
00637 
00638       if (common_packet_format_data_item->data_item.type_id
00639           == kCipItemIdConnectedDataItem) { /* Connected Item */
00640         message_size = EncodeConnectedDataItemLength(message_router_response,
00641                                                      &message, message_size);
00642         message_size = EncodeSequenceNumber(message_size,
00643                                             &g_common_packet_format_data_item,
00644                                             &message);
00645 
00646       } else { /* Unconnected Item */
00647         message_size = EncodeUnconnectedDataItemLength(message_size,
00648                                                        message_router_response,
00649                                                        &message);
00650       }
00651 
00652       /* write message router response into linear memory */
00653       message_size = EncodeReplyService(message_size, &message,
00654                                         message_router_response);
00655       message_size = EncodeReservedFieldOfLengthByte(message_size, &message,
00656                                                      message_router_response);
00657       message_size = EncodeGeneralStatus(message_size, &message,
00658                                          message_router_response);

00660                                           message_router_response);
00661       message_size = EncodeMessageRouterResponseData(message_size,
00662                                                      message_router_response,
00663                                                      &message);
00664     } else { /* connected IO Message to send */
00665       message_size = EncodeDataItemType(common_packet_format_data_item,
00666                                         &message, message_size);
00667 
00668       message_size = EncodeDataItemLength(common_packet_format_data_item,
00669                                           &message, message_size);
00670 
00671       message_size = EncodeDataItemData(common_packet_format_data_item,
00672                                         &message, message_size);

00674   }

00676   /* process SockAddr Info Items */
00677   /* make sure first the O->T and then T->O appears on the wire.
00678    * EtherNet/IP specification doesn't demand it, but there are EIP
00679    * devices which depend on CPF items to appear in the order of their
00680    * ID number */
00681   for (int type = kCipItemIdSocketAddressInfoOriginatorToTarget;
00682        type <= kCipItemIdSocketAddressInfoTargetToOriginator; type++) {

00690                                                 common_packet_format_data_item,
00691                                                 &message);

00694           common_packet_format_data_item->address_info_item[j].sin_port,
00695           common_packet_format_data_item->address_info_item[j].sin_addr,
00696           &message);
00697 
00698         message_size += FillNextNMessageOctetsWithValueAndMoveToNextPosition(
00699           0, 8, &message);
00700         break;
00701       }
00702     }
00703   }
