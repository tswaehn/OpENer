326 EipStatus GetAttributeAllTcpIpInterface(
327  CipInstance *instance,
328  CipMessageRouterRequest *message_router_request,
329  CipMessageRouterResponse *message_router_response,
330  struct sockaddr *originator_address,
331  const int encapsulation_session) {
332 
333  EipUint8 *response = message_router_response->data; /* pointer into the reply */
334  CipAttributeStruct *attribute = instance->attributes;
335 
336  for (int j = 0; j < instance->cip_class->number_of_attributes; j++) /* for each instance attribute of this class */
337  {
338  int attribute_number = attribute->attribute_number;
339  if (attribute_number < 32) /* only return attributes that are flagged as being part of GetAttributeALl */
340  {
341  message_router_request->request_path.attribute_number = attribute_number;
342 
343  if (8 == attribute_number) { /* insert 6 zeros for the required empty safety network number according to Table 5-3.10 */
344  memset(message_router_response->data, 0, 6);
345  message_router_response->data += 6;
346  }
347 
348  if ( kEipStatusOkSend
349  != GetAttributeSingleTcpIpInterface(instance, message_router_request,
350  message_router_response,
351  originator_address,
352  encapsulation_session) ) {
353  message_router_response->data = response;
354  return kEipStatusError;
355  }
356  message_router_response->data += message_router_response->data_length;
357 
358  if (9 == attribute_number) {
359  /* returning default value for unimplemented attributes 10,11 and 12 */
360 
361  /* attribute 10, type: BOOL, default value: 0 */
362  message_router_response->data += 6;
363  *(message_router_response->data) = 0;
364  message_router_response->data += 1;
365 
366  /* attribute 11, type: STRUCT OF USINT, ARRAY of 6 USINTs, ARRAY of 28 USINTs default value: 0 */
367  memset(message_router_response->data, 0, 29);
368  message_router_response->data += 29;
369 
370  /* attribute 12, type: BOOL default value: 0 */
371  *(message_router_response->data) = 0;
372  message_router_response->data += 1;
373  }
374 
375  }
376  attribute++;
377  }
378  message_router_response->data_length = message_router_response->data
379  - response;
380  message_router_response->data = response;
381 
382  return kEipStatusOkSend;
383 }
